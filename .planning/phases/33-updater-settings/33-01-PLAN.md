const { settingsFile } = require('../utils/paths');
function loadSettings() {
  try {
    if (!fs.existsSync(settingsFile)) return null;
    const stat = fs.statSync(settingsFile);
    if (cachedSettings && stat.mtimeMs === settingsMtime) return cachedSettings;
    cachedSettings = JSON.parse(fs.readFileSync(settingsFile, 'utf8'));
    settingsMtime = stat.mtimeMs;
    return cachedSettings;
  } catch { return null; }
}

**2. Update `initialize()` to be settings-aware — CRITICAL: move autoDownload/autoInstallOnAppQuit OUTSIDE the isInitialized guard:**

The `initialize()` method has an `if (this.isInitialized) return;` guard at the top. The event listeners and one-time setup should stay inside that guard, but `autoDownload` and `autoInstallOnAppQuit` must be re-read every time so that settings changes take effect without restarting the app.

Restructure `initialize()` like this:
```javascript
initialize() {
  // Always re-read settings for download/install mode (user may change mid-session)
  const settings = this.loadSettings();
  const downloadMode = settings.updateDownloadMode || 'auto';
  const installMode = settings.updateInstallMode || 'auto';
  autoUpdater.autoDownload = (downloadMode === 'auto');
  autoUpdater.autoInstallOnAppQuit = (installMode === 'auto');

  if (this.isInitialized) return;

  // ... rest of one-time initialization (event listeners, etc.) ...

  this.isInitialized = true;
}
```

This ensures that even after `isInitialized` is true, calling `initialize()` (via `checkForUpdates()`) will re-read the current settings and apply them.

**4. Update `startPeriodicCheck()` to use configured interval:**

```javascript
startPeriodicCheck() {
  if (this.checkInterval) {
    clearInterval(this.checkInterval);
    this.checkInterval = null;
  }

  const settings = this.loadSettings();
  const intervalKey = settings.updateCheckInterval || '30min';
  const intervalMs = CHECK_INTERVALS[intervalKey];

  // startup-only or manual = no periodic checks
  if (!intervalMs) return;

  this.checkInterval = setInterval(() => {
    if (!this.isDownloading) {
      console.debug('Periodic update check...');
      autoUpdater.checkForUpdates().catch(err => {
        console.error('Periodic update check failed:', err);
      });
    }
  }, intervalMs);
}
```
