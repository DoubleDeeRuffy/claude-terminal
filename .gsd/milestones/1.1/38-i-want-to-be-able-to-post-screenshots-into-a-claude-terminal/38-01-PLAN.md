### B. Per-terminal image state functions

Add these functions after `performPaste`:

```javascript
function getTerminalPendingImages(terminalId) {
  const td = getTerminal(terminalId);
  if (!td) return [];
  if (!td.pendingImages) td.pendingImages = [];
  return td.pendingImages;
}

function addTerminalImage(terminalId, imageData) {
  const pending = getTerminalPendingImages(terminalId);
  if (pending.length >= MAX_TERMINAL_IMAGES) return false;
  pending.push(imageData);
  renderTerminalImagePreview(terminalId);
  return true;
}

function removeTerminalImage(terminalId, index) {
  const pending = getTerminalPendingImages(terminalId);
  if (index < 0 || index >= pending.length) return;
  const removed = pending.splice(index, 1)[0];
  if (removed?.objectUrl) URL.revokeObjectURL(removed.objectUrl);
  renderTerminalImagePreview(terminalId);
}

function clearTerminalImages(terminalId) {
  const pending = getTerminalPendingImages(terminalId);
  pending.forEach(img => { if (img.objectUrl) URL.revokeObjectURL(img.objectUrl); });
  pending.length = 0;
  renderTerminalImagePreview(terminalId);
}
```

### C. Preview bar rendering

```javascript
function renderTerminalImagePreview(terminalId) {
  const wrapper = document.querySelector(`.terminal-wrapper[data-id="${terminalId}"]`);
  if (!wrapper) return;
  let previewBar = wrapper.querySelector('.terminal-image-preview');
  const pending = getTerminalPendingImages(terminalId);

  if (pending.length === 0) {
    if (previewBar) {
      previewBar.style.display = 'none';
      previewBar.innerHTML = '';
    }
    triggerTerminalFit(terminalId);
    return;
  }

  // Create preview bar if it doesn't exist
  if (!previewBar) {
    previewBar = document.createElement('div');
    previewBar.className = 'terminal-image-preview';
    previewBar.setAttribute('aria-live', 'polite');
    // Insert before the xterm element (first child that is .xterm or after loading overlay)
    const xtermEl = wrapper.querySelector('.xterm');
    if (xtermEl) {
      wrapper.insertBefore(previewBar, xtermEl);
    } else {
      wrapper.appendChild(previewBar);
    }
  }

  previewBar.style.display = 'flex';
  previewBar.innerHTML = pending.map((img, i) => `
    <div class="terminal-image-thumb" data-index="${i}">
      <img src="${img.objectUrl || img.dataUrl}" alt="Screenshot ${i + 1}" />
      <button class="terminal-image-remove" data-index="${i}" tabindex="0" role="button" aria-label="${t('terminals.imagePreview.remove', { n: i + 1 })}">&times;</button>
    </div>
  `).join('') + (pending.length >= 2 ? `<span class="terminal-image-count">${pending.length}/${MAX_TERMINAL_IMAGES}</span>` : '');

  previewBar.querySelectorAll('.terminal-image-remove').forEach(btn => {
    btn.addEventListener('click', (e) => {
      e.stopPropagation();
      removeTerminalImage(terminalId, parseInt(btn.dataset.index));
    });
  });

  triggerTerminalFit(terminalId);
}

function triggerTerminalFit(terminalId) {
  const td = getTerminal(terminalId);
  if (td?.fitAddon) {
    requestAnimationFrame(() => td.fitAddon.fit());
  }
}
```

### D. Paste interception for images

Add `handleTerminalImagePaste` function:

```javascript
function handleTerminalImagePaste(terminalId, imageItems) {
  const pending = getTerminalPendingImages(terminalId);
  for (const item of imageItems) {
    if (pending.length >= MAX_TERMINAL_IMAGES) break;
    if (!SUPPORTED_IMAGE_TYPES.includes(item.type)) continue;
    const file = item.getAsFile();
    if (!file || file.size > MAX_IMAGE_SIZE) continue;
    const objectUrl = URL.createObjectURL(file);
    const reader = new FileReader();
    reader.onload = () => {
      if (pending.length >= MAX_TERMINAL_IMAGES) {
        URL.revokeObjectURL(objectUrl);
        return;
      }
      const dataUrl = reader.result;
      const base64 = dataUrl.split(',')[1];
      pending.push({ base64, mediaType: file.type, name: `screenshot-${Date.now()}`, dataUrl, objectUrl });
      renderTerminalImagePreview(terminalId);
    };
    reader.readAsDataURL(file);
  }
}
```

### A. Temp file save function (TerminalManager.js, near the image state functions)

```javascript
function saveImagesToTemp(pendingImages) {
  const { fs: nodeFs, path: nodePath, os } = window.electron_nodeModules;
  const tempDir = nodePath.join(os.homedir(), '.claude-terminal', 'temp');
  nodeFs.mkdirSync(tempDir, { recursive: true });
  const timestamp = Date.now();
  const paths = [];
  for (let i = 0; i < pendingImages.length; i++) {
    const img = pendingImages[i];
    const filePath = nodePath.join(tempDir, `screenshot-${timestamp}-${i}.png`);
    const buffer = Buffer.from(img.base64, 'base64');
    nodeFs.writeFileSync(filePath, buffer);
    paths.push(filePath);
  }
  return paths;
}
```

### B. Modify `terminal.onData` handler (around line 1747)

Replace the current `terminal.onData(data => {` block. The key change: when `data === '\r'` and there are pending images, intercept the Enter to inject file paths before sending to PTY.

The modified handler (preserving all existing behavior):

```javascript
terminal.onData(data => {
  const td = getTerminal(id);

  // Enter key with pending images — inject file paths (per D-04)
  if ((data === '\r' || data === '\n') && td?.pendingImages?.length > 0) {
    try {
      const paths = saveImagesToTemp(td.pendingImages);
      // Append file paths to the PTY input, then send Enter
      // The user's typed text (inputBuffer) is already echoed char-by-char.
      // We append a space + each file path, then \r.
      const pathsStr = paths.map(p => `"${p.replace(/\\/g, '/')}"`).join(' ');
      api.terminal.input({ id, data: ` ${pathsStr}\r` });
    } catch (err) {
      console.error('[TerminalManager] Failed to save images to temp:', err);
      // Fallback: just send Enter without images
      api.terminal.input({ id, data });
    }
    clearTerminalImages(id);
    // Continue with normal Enter handling below (status update, title extract, etc.)
    if (td?.project?.id) heartbeat(td.project.id, 'terminal');
    cancelScheduledReady(id);
    updateTerminalStatus(id, 'working');
    if (scrapingEventCallback) scrapingEventCallback(id, 'input', {});
    if (td && td.inputBuffer.trim().length > 0) {
      postEnterExtended.add(id);
      const title = extractTitleFromInput(td.inputBuffer);
      if (title) updateTerminalTabName(id, title);
      updateTerminal(id, { inputBuffer: '' });
    }
    return;
  }

  // Normal input — send to PTY as before
  api.terminal.input({ id, data });
  if (td?.project?.id) heartbeat(td.project.id, 'terminal');
  if (data === '\r' || data === '\n') {
    cancelScheduledReady(id);
    updateTerminalStatus(id, 'working');
    if (scrapingEventCallback) scrapingEventCallback(id, 'input', {});
    if (td && td.inputBuffer.trim().length > 0) {
      postEnterExtended.add(id);
      const title = extractTitleFromInput(td.inputBuffer);
      if (title) updateTerminalTabName(id, title);
      updateTerminal(id, { inputBuffer: '' });
    }
  } else if (data === '\x7f' || data === '\b') {
    if (td) updateTerminal(id, { inputBuffer: td.inputBuffer.slice(0, -1) });
  } else if (data.length === 1 && data.charCodeAt(0) >= 32) {
    if (td) updateTerminal(id, { inputBuffer: td.inputBuffer + data });
  }
});
```

**CRITICAL:** The file paths are forward-slashed (`.replace(/\\/g, '/')`) and quoted for Windows compatibility. Claude Code CLI accepts forward slashes on Windows.
