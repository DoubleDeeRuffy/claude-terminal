2. Create the `handleAiRename(id)` function near `showTabContextMenu`:

```javascript
async function handleAiRename(id) {
  const termData = getTerminal(id);
  if (!termData) return;

  // Save original name for revert on failure
  const originalName = termData.name || '';

  // Show loading indicator immediately
  updateTerminalTabName(id, '...');

  try {
    // Use terminal's current name as input (same source as automatic AI naming)
    const input = originalName || 'terminal';
    const res = await api.chat.generateTabName({ userMessage: input });
    if (res?.success && res.name) {
      updateTerminalTabName(id, res.name);
    } else {
      // Revert on failure
      updateTerminalTabName(id, originalName);
    }
  } catch (err) {
    // Revert on error/timeout
    updateTerminalTabName(id, originalName);
  }
}
```
